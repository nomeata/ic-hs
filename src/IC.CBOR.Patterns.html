pattern TNat :: Natural -> Term
pattern $mTNat :: forall r. Term -> (Natural -> r) -> (Void# -> r) -> r

        TInt m | m >= 0 -> Just (fromIntegral m)
        TInteger m | m >= 0 -> Just (fromIntegral m)
        _ -> Nothing
    -> Just m)
